        // Application State
        let currentData = [];
        let sortConfig = { key: 'price', direction: 'desc' };
        let searchTerm = '';
        let selectedZone = 'All';

        // DOM Elements
        const tableBody = document.getElementById('tableBody');
        const searchInput = document.getElementById('searchInput');
        const filterButtons = document.getElementById('filterButtons');
        const exportBtn = document.getElementById('exportBtn');
        const sortDateIcon = document.getElementById('sortDateIcon');
        const sortCentreIcon = document.getElementById('sortCentreIcon');
        const sortPriceIcon = document.getElementById('sortPriceIcon');
        const recordCount = document.getElementById('record-count');
        const showingCount = document.getElementById('showing-count');
        const totalRecords = document.getElementById('total-records');
        const avgPrice = document.getElementById('avg-price');
        const maxPrice = document.getElementById('max-price');
        const minPrice = document.getElementById('min-price');
        const totalCount = document.getElementById('total-count');

        // Initialize
        function init() {
            // Process initial data
            currentData = processData(apiResponse.data);
            
            // Initialize filter buttons
            initFilterButtons();
            
            // Render table
            renderTable();
            
            // Update statistics
            updateStatistics();
            
            // Set up event listeners
            setupEventListeners();
        }

        // Process raw data
        function processData(rawData) {
            return rawData.map(row => {
                const [date, zone, centre, price] = row;
                const priceNum = parseInt(price);
                
                // Format date
                const [day, month, year] = date.split('-');
                const formattedDate = `${day}/${month}/${year}`;
                
                return {
                    date,
                    zone,
                    centre,
                    price: priceNum,
                    formattedDate
                };
            });
        }

        // Initialize filter buttons
        function initFilterButtons() {
            const zones = ['All', ...new Set(apiResponse.data.map(row => row[1]))];
            
            filterButtons.innerHTML = zones.map(zone => {
                const isActive = zone === 'All';
                return `
                    <button class="filter-btn ${isActive ? 'active' : ''}" data-zone="${zone}">
                        ${zone === 'All' ? '<i class="fas fa-filter"></i> All Zones' : `Zone ${zone}`}
                    </button>
                `;
            }).join('');
        }

        // Get filtered and sorted data
        function getProcessedData() {
            let result = [...currentData];
            
            // Apply search filter
            if (searchTerm) {
                const term = searchTerm.toLowerCase();
                result = result.filter(item =>
                    item.centre.toLowerCase().includes(term) ||
                    item.zone.toString().includes(term)
                );
            }
            
            // Apply zone filter
            if (selectedZone !== 'All') {
                result = result.filter(item => item.zone === selectedZone);
            }
            
            // Apply sorting
            result.sort((a, b) => {
                if (sortConfig.key === 'price') {
                    return sortConfig.direction === 'asc' 
                        ? a.price - b.price 
                        : b.price - a.price;
                } else if (sortConfig.key === 'centre') {
                    return sortConfig.direction === 'asc'
                        ? a.centre.localeCompare(b.centre)
                        : b.centre.localeCompare(a.centre);
                } else if (sortConfig.key === 'date') {
                    return sortConfig.direction === 'asc'
                        ? a.date.localeCompare(b.date)
                        : b.date.localeCompare(a.date);
                }
                return 0;
            });
            
            return result;
        }

        // Render table
        function renderTable() {
            const data = getProcessedData();
            
            if (data.length === 0) {
                tableBody.innerHTML = `
                    <tr>
                        <td colspan="4" class="no-results">
                            <i class="fas fa-search fa-lg" style="margin-bottom: 10px;"></i><br>
                            No matching records found
                        </td>
                    </tr>
                `;
            } else {
                tableBody.innerHTML = data.map(item => {
                    const priceClass = item.price > 700 ? 'price-high' : 
                                      item.price < 550 ? 'price-low' : 'price-normal';
                    
                    return `
                        <tr>
                            <td>
                                <div class="date-cell">
                                    <span class="date-formatted">${item.formattedDate}</span>
                                    <span class="date-original">${item.date}</span>
                                </div>
                            </td>
                            <td>
                                <span class="zone-badge zone-${item.zone}">
                                    Zone ${item.zone}
                                </span>
                            </td>
                            <td>${item.centre}</td>
                            <td>
                                <div class="price-cell">
                                    <span class="${priceClass}">₹${item.price}</span>
                                    ${item.price > 700 ? '<span class="price-badge high-badge">High</span>' : 
                                      item.price < 550 ? '<span class="price-badge low-badge">Low</span>' : ''}
                                </div>
                            </td>
                        </tr>
                    `;
                }).join('');
            }
            
            // Update counts
            showingCount.textContent = data.length;
            totalRecords.textContent = currentData.length;
            recordCount.textContent = currentData.length;
        }

        // Update statistics
        function updateStatistics() {
            const data = getProcessedData();
            
            if (data.length > 0) {
                const prices = data.map(d => d.price);
                const avg = Math.round(prices.reduce((a, b) => a + b, 0) / prices.length);
                const max = Math.max(...prices);
                const min = Math.min(...prices);
                
                avgPrice.textContent = `₹${avg}`;
                maxPrice.textContent = `₹${max}`;
                minPrice.textContent = `₹${min}`;
                totalCount.textContent = data.length;
            } else {
                avgPrice.textContent = '₹0';
                maxPrice.textContent = '₹0';
                minPrice.textContent = '₹0';
                totalCount.textContent = '0';
            }
        }

        // Update sort icons
        function updateSortIcons() {
            // Reset all icons
            sortDateIcon.innerHTML = '';
            sortCentreIcon.innerHTML = '';
            sortPriceIcon.innerHTML = '';
            
            // Set active sort icon
            const icon = sortConfig.direction === 'asc' ? '↑' : '↓';
            
            if (sortConfig.key === 'date') {
                sortDateIcon.innerHTML = icon;
            } else if (sortConfig.key === 'centre') {
                sortCentreIcon.innerHTML = icon;
            } else if (sortConfig.key === 'price') {
                sortPriceIcon.innerHTML = icon;
            }
        }

        // Setup event listeners
        function setupEventListeners() {
            // Search input
            searchInput.addEventListener('input', (e) => {
                searchTerm = e.target.value;
                renderTable();
                updateStatistics();
            });
            
            // Filter buttons
            filterButtons.addEventListener('click', (e) => {
                if (e.target.classList.contains('filter-btn')) {
                    const zone = e.target.dataset.zone;
                    selectedZone = zone;
                    
                    // Update active button
                    document.querySelectorAll('.filter-btn').forEach(btn => {
                        btn.classList.remove('active');
                    });
                    e.target.classList.add('active');
                    
                    renderTable();
                    updateStatistics();
                }
            });
            
            // Sort headers
            document.querySelectorAll('th[data-sort]').forEach(th => {
                th.addEventListener('click', () => {
                    const key = th.dataset.sort;
                    
                    if (sortConfig.key === key) {
                        // Toggle direction if same key
                        sortConfig.direction = sortConfig.direction === 'asc' ? 'desc' : 'asc';
                    } else {
                        // Set new key with default direction
                        sortConfig.key = key;
                        sortConfig.direction = 'desc';
                    }
                    
                    updateSortIcons();
                    renderTable();
                });
            });
            
            // Export button
            exportBtn.addEventListener('click', () => {
                const data = getProcessedData();
                const csvContent = [
                    ['Date', 'Zone', 'Centre', 'Price'].join(','),
                    ...data.map(item => [item.date, item.zone, item.centre, item.price].join(','))
                ].join('\n');
                
                const blob = new Blob([csvContent], { type: 'text/csv' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'crop-prices.csv';
                a.click();
                URL.revokeObjectURL(url);
                
                // Show feedback
                const originalText = exportBtn.innerHTML;
                exportBtn.innerHTML = '<i class="fas fa-check"></i> Exported!';
                setTimeout(() => {
                    exportBtn.innerHTML = originalText;
                }, 2000);
            });
        }

        document.addEventListener('DOMContentLoaded', init);
